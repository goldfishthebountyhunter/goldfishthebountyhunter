#!/bin/bash
echo  “Starting Install”
sleep 1
echo  “Initiating...”

# This script tries to install packages from an array
# with a function that searches for each linux distros package managers.

# Array of packages to install
packages=("wget" "curl" "git")  # Add more packages to this list as needed

# Function to install packages
install_packages() {
    local package_manager="$1"
    local install_command="$2"

    echo "Attempting to install packages using $package_manager..."
    for package in "${packages[@]}"; do
        if $install_command $package; then
            echo "Successfully installed $package using $package_manager."
        else
            echo "Failed to install $package using $package_manager."
        fi
    done
}

# Check and install packages using various package managers

if command -v apt-get &> /dev/null; then
    install_packages "apt-get" "sudo apt-get install -y"
elif command -v dnf &> /dev/null; then
    install_packages "dnf" "sudo dnf install -y"
elif command -v yum &> /dev/null; then
    install_packages "yum" "sudo yum install -y"
elif command -v zypper &> /dev/null; then
    install_packages "zypper" "sudo zypper install -y"
elif command -v pacman &> /dev/null; then
    install_packages "pacman" "sudo pacman -Syu --noconfirm"
elif command -v apk &> /dev/null; then
    install_packages "apk" "sudo apk add"
elif command -v brew &> /dev/null; then
    install_packages "brew" "brew install"
elif command -v snap &> /dev/null; then
    for package in "${packages[@]}"; do
        if sudo snap install "$package"; then
            echo "Successfully installed $package using snap."
        else
            echo "Failed to install $package using snap."
        fi
    done
else
    echo "No recognized package manager found. Please install the packages manually."
    exit 1
fi

echo "Package installation attempts complete."
exit 0

# Launch another script after this one completes
# ./another_script.sh  # Replace this with the path to the script you want to run
